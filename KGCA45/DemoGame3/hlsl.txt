//--------------------------------------------------------------------------------------
// Vertex Shader
//--------------------------------------------------------------------------------------
float4 VS( in float2 p : POSITION ) : SV_POSITION
{
    return float4(p.x, p.y, 0.0f, 1.0f);
}

//--------------------------------------------------------------------------------------
// Pixel Shader
//--------------------------------------------------------------------------------------
float4 PS( in float4 p : SV_POSITION ) : SV_Target
{	
	return float4(1,1,0,1);
}
